150 void tsc_measure(uint32_t chan_io)
151 {
152  /*##-2- Discharge the touch-sensing IOs ##################################*/
153  /* Must be done before each acquisition */
154  __HAL_TSC_CLOSE_ANALOG_SWITCH(&htsc, TSC_GROUP6_IO3|TSC_GROUP6_IO2|TSC_GROUP6_IO4);
155  __HAL_TSC_SET_IODEF_INFLOAT(&htsc);
156  HAL_Delay(1); /* 1 ms is more than enough to discharge all capacitors */
157 
158  /* ##-3- Config IO channel ############################*/
159  TSC_IOConfigTypeDef tsc_io_config;
160  tsc_io_config.ChannelIOs = TSC_Channel_IO[chan_io];
161  tsc_io_config.SamplingIOs = htsc.Init.SamplingIOs;
162  tsc_io_config.ShieldIOs = htsc.Init.ShieldIOs;
163 
164  HAL_TSC_IOConfig(&htsc, &tsc_io_config);
165 
166  if (HAL_TSC_Start(&htsc) != HAL_OK)
167  {
168  Error_Handler();
169  }
170 
171  /*##-4- Wait for the end of acquisition ##################################*/
172  /* Before starting a new acquisition, you need to check the current state of
173  the peripheral; if it’s busy you need to wait for the end of current
174  acquisition before starting a new one. */
175  while (HAL_TSC_GetState(&htsc) == HAL_TSC_STATE_BUSY)
176  {
177  /* For simplicity reasons, this example is just waiting till the end of the
178  acquisition, but application may perform other tasks while acquisition
179  operation is ongoing. */
180  }
181 
182  /*##-5- Clear flags ######################################################*/
183  __HAL_TSC_CLEAR_FLAG(&htsc, (TSC_FLAG_EOA | TSC_FLAG_MCE));
184 
185  /*##-6- Check if the acquisition is correct (no max count) ###############*/
186  if (HAL_TSC_GroupGetStatus(&htsc, TSC_GROUP6_IDX) == TSC_GROUP_COMPLETED)
187  {
188  /*##-7- Store the acquisition value ####################################*/
189  TSC_acquisition_value[chan_io] = HAL_TSC_GroupGetValue(&htsc, TSC_GROUP6_IDX);
190  }
