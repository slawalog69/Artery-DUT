109 // Применяется табличная коррекция
110 static int32_t calc_corr_cnt_by_temperature(uint32_t cnt, int8_t iTemp, uint8_t chan)
111 {
112  int16_t corr_val;
113  int16_t const *p_array;
114 
115  if (chan == 1) p_array = array_cnt_ref_corr;
116  else if (chan == 2) p_array = array_cnt_main_corr;
117  else return (cnt);
118 
119  // Проверим на крайние значения
120  if (iTemp <= TEMP_CORR_MIN)
121  {
122  corr_val = *(uint16_t*)p_array;
123  }
124  else if (iTemp >= TEMP_CORR_MAX)
125  {
126  corr_val = *(uint16_t*)(p_array + TEMP_CORR_POINTS - 1);
127  }
128  else
129  {
130  // Если температура не выходит за крайние значения определим номер для корректировки
131  int8_t i = (iTemp - TEMP_CORR_MIN) / TEMP_CORR_STEP; // номер индекса элемента в массиве коррекции
132  int16_t corr_A, corr_B;
133  corr_A = *(uint16_t*)(p_array + i); // значение коррекции array[i]
134  corr_B = *(uint16_t*)(p_array + i + 1); // значение коррекции array[i + 1]
135  // т.к. шаг температурной коррекции составляет более одного градуса, то для сглаживания между шагами коррекции необходимо
136  // учесть значение температуры: для этого вычисляется остаток от деления температуры на шаг коррекции
137 // corr_val = corr_A + (((iTemp - TEMP_CORR_MIN) % TEMP_CORR_STEP) * (corr_B - corr_A) / TEMP_CORR_STEP); // расчет поправки CNT
138  corr_val = corr_A + (s16)((((float)((iTemp - TEMP_CORR_MIN) % TEMP_CORR_STEP)) * ((float)(corr_B - corr_A)) / TEMP_CORR_STEP));
139  }
140 
141  return (cnt + settings_sensor.multiplier * corr_val); // коррекция
142 }

177 static uint16_t measurement_update_filter(uint16_t value)
178 {
179 
180  // ** допуск контроль settings_sensor.filter_length и его мультипликация
181  uint16_t filter_length = ((settings_sensor.filter_length > 30 ) ? MAX_VAR_FILTER_LENGTH : (settings_sensor.filter_length*10)); //**
182  // 300 // max 30 *10
183 
184  if (settings_sensor.filter_length > 1 )
185  {
186  filter_length+=MAX_CONST_FILTER_LENGTH; //**добавляем постоянное окно фильтрации MAX_CONST_FILTER_LENGTH =150 (~ 20 сек)
187  }
188  else
189  {
190  filter_length=30; // макс глубина фильтра при отключенной фильтрации
191  }
192 
193 // uint16_t m = (filter_length - 1)/2; //** определяем смещение медианного значения
194 
195  meas_state.levels[meas_state.measurement_counter] = value; // ** добавляем значение в фильтр циклически с перезаписью
196 
197  if (meas_state.filter_started == 1) //** Если фильтр заполненн полностью можно фильтровать
198  {
199  /* Начало постоянной фильтрации */
200  //***********************************************************************************************
201  uint32_t avg_value = 0;
202 
203  // максимальная***********************************************************************
204 
205  if (settings_sensor.filter_length > 29 )
206  {
207  int16_t spos=0;
208  uint16_t avrange=20;
209  uint16_t vdiv=0;
210 
211 
212  spos=meas_state.measurement_counter - avrange;
213  if (spos < 0 )
214  {
215  spos+=filter_length;
216  }
217 
218  int i = 0;
219 
220  while (i < avrange)
221  {
222  avg_value+=(meas_state.levels[spos]*(i+1));
223  vdiv+=(i+1);
224  i++;
225  spos++;
226  if (spos > (filter_length-1))
227  {
228  spos=1;
229  }
230  }
231  meas_state.levels[meas_state.measurement_counter] = (uint16_t)((double)avg_value / (double)vdiv);
232  }
233 
234  //средняя и ниже ********************************************************************************************
235 
236  avg_value = 0;
237 
238  for (int i = 1; i < (filter_length); i++) //** Перебираем все значения в фильтре
239  {
240  avg_value += meas_state.levels[i]; //** суммируем все значения фильтра
241  }
242  meas_state.filtered = (uint16_t)(avg_value / (filter_length-1));
243 
244  /*************************************************************************/
245 
246  }
247  else
248  {
249  // measure_temperature();
250  if (meas_state.measurement_counter == 0)
251  meas_state.filtered = value;
252 
253  if (meas_state.measurement_counter == 1)
254  meas_state.levels[0] = meas_state.levels[1];
255 
256  uint32_t avg_value = 0;
257  uint16_t vdiv = meas_state.measurement_counter + 1;
258 
259  for (int i = 0; i <= meas_state.measurement_counter; i++)
260  {
261  avg_value += meas_state.levels[i];
262  }
263  meas_state.filtered = (uint16_t)(avg_value / vdiv);
264  }
265 
266  meas_state.measurement_counter++;
267 
268  if ( (meas_state.measurement_counter >= (filter_length - 1)) && (meas_state.filter_started == 0) ) {
269  meas_state.filter_started = 1;
270  meas_state.cnt_t_last_prd_chk = meas_state.chan_main.cnt_curr;
271  }
272 
273  meas_state.measurement_counter %= filter_length;
274 
275  return meas_state.filtered;
276

322 static void calc_levels(int32_t cnt)
323 {
324  if (cnt < meas_state.chan_main.cnt_empty)
325  {
326  cnt = meas_state.chan_main.cnt_empty;
327  }
328  else if (cnt > meas_state.chan_main.cnt_full)
329  {
330  cnt = meas_state.chan_main.cnt_full;
331  }
332 
333  uint32_t n_diff = (settings_sensor.n_max - settings_sensor.n_min);
334  uint32_t cnt_diff = (meas_state.chan_main.cnt_full - meas_state.chan_main.cnt_empty) * 100;
335  uint32_t a = ((n_diff != 0) && (cnt_diff != 0)) ? (cnt_diff / n_diff) : 100;
336  uint32_t cnt_relative = (cnt - meas_state.chan_main.cnt_empty) * 100;
337  uint32_t t = cnt_relative / a + settings_sensor.n_min;
338  wdt_refresh();//*
339  meas_state.current_level = t;
340  meas_state.relative_level = measurement_update_filter((uint16_t)t);
341  wdt_refresh();//*
342  if (meas_state.relative_level < settings_sensor.n_min)
343  {
344  meas_state.relative_level = settings_sensor.n_min;
345  }
346  else if (meas_state.relative_level > settings_sensor.n_max)
347  {
348  meas_state.relative_level = settings_sensor.n_max;
349  }
350 
351  if (meas_state.current_level < settings_sensor.n_min)
352  {
353  meas_state.current_level = settings_sensor.n_min;
354  }
355  else if (meas_state.current_level > settings_sensor.n_max)
356  {
357  meas_state.current_level = settings_sensor.n_max;
358  }
359  // вычислим значения регистров для modbus
360  // mb_hold_reg.relative_level = ((n_diff > 0) && (meas_state.relative_level > 0)) ? (uint16_t)(meas_state.relative_level / (n_diff / 1000)) : 0;
361  mb_hold_reg.relative_level = meas_state.relative_level;
362  mb_calc_hold_reg_coef_corr();
363 }

649  int8_t measure_calc_corr_temperature()
650 {
651  int8_t corr_val;
652  int8_t const *p_array = array_temperature_corr;
653  int8_t iTemp = meas_state.temperature_pure;
654 
655  // Проверим на крайние значения
656  if (iTemp <= TEMP_CORR_MIN)
657  {
658  corr_val = *(int8_t*)p_array;
659  }
660  else if (iTemp >= TEMP_CORR_MAX)
661  {
662  corr_val = *(int8_t*)(p_array + TEMP_CORR_POINTS - 1);
663  }
664  else
665  {
666  // Если температура не выходит за крайние значения определим номер для корректировки
667  int8_t i = (iTemp - TEMP_CORR_MIN) / TEMP_CORR_STEP; // номер индекса элемента в массиве коррекции
668  int8_t corr_A, corr_B;
669  corr_A = *(int8_t*)(p_array + i); // значение коррекции array[i]
670  corr_B = *(int8_t*)(p_array + i + 1); // значение коррекции array[i + 1]
671  // т.к. шаг температурной коррекции составляет более одного градуса,
672  //то для сглаживания между шагами коррекции необходимо
673  // учесть значение температуры: для этого вычисляется
674  //остаток от деления температуры на шаг коррекции
675  corr_val = corr_A + (s8)((((iTemp - TEMP_CORR_MIN) % TEMP_CORR_STEP) * ((float)(corr_B - corr_A)) / TEMP_CORR_STEP)); // расчет поправки CNT
676 // corr_val = corr_A + (((iTemp - (-40)) % 10) * (float)(corr_B - corr_A) / 10);
677 
678  }
679 
680  return (iTemp + corr_val);
681 }

683 static void measure_calc_delta_between_channels()
684 {
685  const uint8_t MEAS_DELTA_QNT = 100;
686  static int32_t cnt_chan_delta = 0;
687  static uint32_t cnt_head_value = 0;
688  static uint8_t n_meas = 0;
689  static uint8_t dummy_meas = 0;
690  static uint8_t flg_cnt_head = 0;
691 
692  if (++dummy_meas < 2)
693  {
694  return; // пропуск измерений
695  }
696 
697  if (n_meas < MEAS_DELTA_QNT)
698  {
699  if (!flg_cnt_head)
700  {
701  cnt_chan_delta += (meas_state.chan_main.cnt_pure - meas_state.chan_ref.cnt_pure);
702  }
703  else
704  {
705  cnt_head_value += (meas_state.chan_main.cnt_curr + meas_state.chan_ref.cnt_curr);
706  }
707  n_meas++;
708  }
709 
710  if (n_meas == MEAS_DELTA_QNT)
711  {
712  settings_sensor_t tmp_settings;
713  tmp_settings = settings_sensor;
714 
715  if (!flg_cnt_head)
716  {
717  // используется мастшабирование на 256 при делении, чтобы не потерять в точности
718  tmp_settings.cnt_chan_delta = (256 * cnt_chan_delta / MEAS_DELTA_QNT) / settings_sensor.multiplier;
719  flg_cnt_head = 1;
720  }
721  else
722  {
723  tmp_settings.cnt_head_value = (256 * cnt_head_value) / (2 * MEAS_DELTA_QNT) / settings_sensor.multiplier;
724  tmp_settings.flg_need_to_delta = 0;
725  meas_state.flg_need_to_calc_delta = 0;
726  flg_cnt_head = 0;
727  }
728  // обновление настроек
729  tmp_settings.crc = crc_calc_CRC8_for_buffer(0, (uint8_t*)&tmp_settings, sizeof(tmp_settings) - 1);
730  settings_sensor_store_data(&tmp_settings);
731  // сброс счетчиков и значений
732  cnt_chan_delta = 0;
733  cnt_head_value = 0;
734  n_meas = 0;
735  dummy_meas = 0;
736  }
