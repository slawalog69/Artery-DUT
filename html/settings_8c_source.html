393  // Modbus states that a baud rate higher than 19200 must use a fixed 750 us
394  // for inter character time out and 1.75 ms for a frame delay.
395  // For baud rates below 19200 the timeing is more critical and has to be calculated.
396  // E.g. 9600 baud in a 10 bit packet is 960 characters per second
397  // In milliseconds this will be 960characters per 1000ms. So for 1 character
398  // 1000ms/960characters is 1.04167ms per character and finaly modbus states an
399  // intercharacter must be 1.5T or 1.5 times longer than a normal character and thus
400  // 1.5T = 1.04167ms * 1.5 = 1.5625ms. A frame delay is 3.5T.
401  // Added experimental low latency delays. This makes the implementation
402  // non-standard but practically it works with all major modbus master implementations.
403 
404 
405  uint32_t table_baudrate[] = { 0, 1200, 2400, 4800, 7200, 9600, 14400, 19200, 38400, 57600, 115200 };
406  u32 baud = table_baudrate[settings_sensor.baudrate];
407 // if ( settings_sensor.baudrate == 1000000)
408  if (baud == 1000000 ){
409  MB_DELAY_EOF = 10;
410  }
411  else if (baud >= 115200){
412  MB_DELAY_EOF = 175;
413  }
414  else if (baud > 19200)
415  {
416  MB_DELAY_EOF = 1750;
417  }
418  else
419  {
420  MB_DELAY_EOF = 35000000/baud; // 1T * 3.5 = T3.5
421  }
