104 void system_core_clock_update(void)
105 {
106  uint32_t pll_mult = 0, pll_mult_h = 0, pll_clock_source = 0, temp = 0, div_value = 0;
107  uint32_t pllrcsfreq = 0, pll_ms = 0, pll_ns = 0, pll_fr = 0;
108  crm_sclk_type sclk_source;
109 
110  static const uint8_t sys_ahb_div_table[16] = {0, 0, 0, 0, 0, 0, 0, 0, 1, 2, 3, 4, 6, 7, 8, 9};
111 
112  /* get sclk source */
113  sclk_source = crm_sysclk_switch_status_get();
114 
115  switch(sclk_source)
116  {
117  case CRM_SCLK_HICK:
118  if(((CRM->misc2_bit.hick_to_sclk) != RESET) && ((CRM->misc1_bit.hickdiv) != RESET))
119  system_core_clock = HICK_VALUE * 6;
120  else
121  system_core_clock = HICK_VALUE;
122  break;
123  case CRM_SCLK_HEXT:
124  system_core_clock = HEXT_VALUE;
125  break;
126  case CRM_SCLK_PLL:
127  pll_clock_source = CRM->cfg_bit.pllrcs;
128  if(CRM->pll_bit.pllcfgen == FALSE)
129  {
130  /* get multiplication factor */
131  pll_mult = CRM->cfg_bit.pllmult_l;
132  pll_mult_h = CRM->cfg_bit.pllmult_h;
133  /* process high bits */
134  if((pll_mult_h != 0U) || (pll_mult == 15U)){
135  pll_mult += ((16U * pll_mult_h) + 1U);
136  }
137  else
138  {
139  pll_mult += 2U;
140  }
141 
142  if (pll_clock_source == 0x00)
143  {
144  /* hick divided by 2 selected as pll clock entry */
145  system_core_clock = (HICK_VALUE >> 1) * pll_mult;
146  }
147  else
148  {
149  /* hext selected as pll clock entry */
150  if (CRM->cfg_bit.pllhextdiv != RESET)
151  {
152  /* hext clock divided by 2 */
153  system_core_clock = (HEXT_VALUE / 2) * pll_mult;
154  }
155  else
156  {
157  system_core_clock = HEXT_VALUE * pll_mult;
158  }
159  }
160  }
161  else
162  {
163  pll_ms = CRM->pll_bit.pllms;
164  pll_ns = CRM->pll_bit.pllns;
165  pll_fr = CRM->pll_bit.pllfr;
166 
167  if (pll_clock_source == 0x00)
168  {
169  /* hick divided by 2 selected as pll clock entry */
170  pllrcsfreq = (HICK_VALUE >> 1);
171  }
172  else
173  {
174  /* hext selected as pll clock entry */
175  if (CRM->cfg_bit.pllhextdiv != RESET)
176  {
177  /* hext clock divided by 2 */
178  pllrcsfreq = (HEXT_VALUE / 2);
179  }
180  else
181  {
182  pllrcsfreq = HEXT_VALUE;
183  }
184  }
185  system_core_clock = (pllrcsfreq * pll_ns) / (pll_ms * (0x1 << pll_fr));
186  }
187  break;
188  default:
189  system_core_clock = HICK_VALUE;
190  break;
191  }
192 
193  /* compute sclk, ahbclk frequency */
194  /* get ahb division */
195  temp = CRM->cfg_bit.ahbdiv;
196  div_value = sys_ahb_div_table[temp];
197  /* ahbclk frequency */
198  system_core_clock = system_core_clock >> div_value;
199 }
