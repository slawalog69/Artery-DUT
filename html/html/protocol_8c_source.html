99 {
100  int i;
101  typeInpPack t;
102  u16 Err = 0xFF;
103  if(usart_get_msg(name,in_buffer, &i, &t) == ERROR_OK){
104  if(t == OMNI_PACK)
105  Err = crc_calc_CRC8_for_buffer(0, in_buffer, i);
106  else if(t == MB_PACK)
107  Err = crc_calc_modbus_CRC16_for_buffer( in_buffer, i);
108  if( Err == ERROR_OK){
109  *pType = t;
110  *p_len = i;
111  return ERROR_OK;
112  }

188 static uint8_t protocol_prepare_tx_buf(uint8_t cmd, uint8_t *p_data, uint8_t len)
189 {
190  int i = 0;
191 
192  // заполнение заголовка
193  out_buffer[i++] = PREFIX_OUTGOING;
194  out_buffer[i++] = device_addr;
195  out_buffer[i++] = cmd;
196 
197  // копирование данных в буфер передачи
198  while (len > 0)
199  {
200  out_buffer[i++] = *p_data++;
201  len--;
202  }
203  out_buffer[i] = crc_calc_CRC8_for_buffer(0, out_buffer, i);
204 
205  return (i + 1);
206 }

215 static uint8_t protocol_modbus_process_request(uart_name_t name, uint8_t len)
216 {
217  uint8_t tx_len = 0;
218 // uint8_t result = RES_SUCCESS;
219 // uint8_t need_to_send_result = 0;
220  uint16_t crc16 = 0;
221  meas_state.last_req = 0;
222  uint8_t mb_function = in_buffer[1];
223  uint16_t startingAddress = ((in_buffer[2] << 8) | in_buffer[3]); // combine the starting address bytes
224  uint16_t no_of_registers = ((in_buffer[4] << 8) | in_buffer[5]); // combine the number of register bytes
225  uint16_t maxData = startingAddress + no_of_registers;
226  uint16_t pdu_start_address = 3; // PDU starts at the 4th byte
227  uint16_t reg_index;
228 
229  switch (mb_function) {
230  case MB_FUNCTION_READ_HOLDING_REGISTERS:
231  // заполним выходной буффер
232  if ((maxData >= MB_MIN_AO_HOLDING_REGS) && (maxData <= MB_MAX_AO_HOLDING_REGS)) // check exception 3 ILLEGAL DATA VALUE
233  {
234  uint8_t noOfBytes = no_of_registers * 2;
235  uint8_t responseFrameSize = 5 + noOfBytes; // ID, function, noOfBytes, (dataLo + dataHi) * number of registers, crcLo, crcHi
236  if (responseFrameSize > FRAME_BUF_LEN) return tx_len = 0;
237  out_buffer[0] = device_addr;
238  out_buffer[1] = mb_function;
239  out_buffer[2] = noOfBytes;
240  for (reg_index = startingAddress; reg_index < maxData; reg_index++)
241  {
242  if (reg_index == settings_sensor. mb_N_reg_address){
243  out_buffer[pdu_start_address++] = mb_hold_reg.relative_level >> 8; // split the register into 2 bytes
244  out_buffer[pdu_start_address++] = mb_hold_reg.relative_level & 0xFF;
245  }
246 
247  else if (reg_index == settings_sensor. mb_T_reg_address){
248  out_buffer[pdu_start_address++] = (uint8_t)((mb_hold_reg.temperature_corr >> 8) & 0xFF); // split the register into 2 bytes
249  out_buffer[pdu_start_address++] = (uint8_t)((mb_hold_reg.temperature_corr >> 0) & 0xFF);
250  } else {
251 
252  return tx_len = 0;
253 
254  }
255  /* off at additional reqauerments
256  if (reg_index == settings_sensor. mb_COEF_reg_address){
257  out_buffer[pdu_start_address++] = (uint8_t)((mb_hold_reg.coef_corr >> 8) & 0xFF); // split the register into 2 bytes
258  out_buffer[pdu_start_address++] = (uint8_t)((mb_hold_reg.coef_corr >> 0) & 0xFF);
259  }
260  */
261  }
262  crc16 = crc_calc_modbus_CRC16_for_buffer(out_buffer, responseFrameSize - 2);
263  out_buffer[responseFrameSize - 2] = crc16 >> 8; // split crc into 2 bytes
264  out_buffer[responseFrameSize - 1] = crc16 & 0xFF;
265  tx_len = responseFrameSize;
266  delay_us(MB_DELAY_EOF);
267  }
268  break; // MB_FUNCTION_READ_HOLDING_REGISTERS
269 
270  case MB_FUNCTION_READ_INPUT_REGISTERS:
271  // заполним выходной буффер
272  if ((maxData >= MB_MIN_AI_INPUT_REGS) && (maxData <= MB_MAX_AI_INPUT_REGS)) // check exception 3 ILLEGAL DATA VALUE
273  {
274  uint8_t noOfBytes = no_of_registers * 2;
275  uint8_t responseFrameSize = 5 + noOfBytes; // ID, function, noOfBytes, (dataLo + dataHi) * number of registers, crcLo, crcHi
276  if (responseFrameSize > FRAME_BUF_LEN) return tx_len = 0;
277  out_buffer[0] = device_addr;
278  out_buffer[1] = mb_function;
279  out_buffer[2] = noOfBytes;
280  for (reg_index = startingAddress; reg_index < maxData; reg_index++)
281  {
282  if (reg_index == settings_sensor. mb_N_reg_address){
283  out_buffer[pdu_start_address++] = mb_hold_reg.relative_level >> 8; // split the register into 2 bytes
284  out_buffer[pdu_start_address++] = mb_hold_reg.relative_level & 0xFF;
285  }
286 
287  if (reg_index == settings_sensor. mb_T_reg_address){
288  out_buffer[pdu_start_address++] = (uint8_t)((mb_hold_reg.temperature_corr >> 8) & 0xFF); // split the register into 2 bytes
289  out_buffer[pdu_start_address++] = (uint8_t)((mb_hold_reg.temperature_corr >> 0) & 0xFF);
290  }
291 
292  if (reg_index == settings_sensor. mb_COEF_reg_address){
293  out_buffer[pdu_start_address++] = (uint8_t)((mb_hold_reg.coef_corr >> 8) & 0xFF); // split the register into 2 bytes
294  out_buffer[pdu_start_address++] = (uint8_t)((mb_hold_reg.coef_corr >> 0) & 0xFF);
295  }
296 
297  }
298  crc16 = crc_calc_modbus_CRC16_for_buffer(out_buffer, responseFrameSize - 2);
299  out_buffer[responseFrameSize - 2] = crc16 >> 8; // split crc into 2 bytes
300  out_buffer[responseFrameSize - 1] = crc16 & 0xFF;
301  tx_len = responseFrameSize;
302  delay_us(MB_DELAY_EOF);
303  }
304  break; //MB_FUNCTION_READ_INPUT_REGISTERS
305  }
306  return tx_len;
307 }
