234 static void fill_ADC_Data(){
235  // static u8 indxTerm = 0;
236 
237  float tmp_mv;
238  // пересчет кода АЦП в напряжение канала VIN
239  // u32 tmp_mv = (adc_dma_buffer[ADC_CHAN_VIN] * VDD_APPLY) >> ADC_VAL_MAX_BR;// / ADC_VAL_DR_MAX;
240  adc_Values[ADC_CHAN_VIN].addVal((float)adc_dma_buffer[ADC_CHAN_VIN]);
241  if(adc_Values[ADC_CHAN_VIN ].calc()){
242  // if(adc_Values[ADC_CHAN_VIN].GetAverage(&tVal) == false) return; // not stable
243  // float tmp_mv = (( tVal * (float)VDD_APPLY))/ (float)ADC_VAL_DR_MAX;// >> ADC_VAL_MAX_BR;//
244  tmp_mv = aprox_Volt(ADC_CHAN_VIN );
245  adc_VIN_mv = (u32)( tmp_mv * (20.0f + 1.0f) / 1.0f *1000.0f);//(10 + 240) / 10; // расчет напряжения на делителе, сопротивление в кОм
246  adc_VIN_mv += 520; //корр на падение напр на диоде
247 
248  if((adc_low_power == 1) &&
249  (adc_VIN_mv > 7000 )) // напряжение на делителе при напряжении на входе 8 Вольт
250  {
251  adc_low_power = 0;
252  // adc_data_ready = 1;
253  } else if ((adc_low_power == 0) &&
254  (adc_VIN_mv < 6000))//if (meas_state.filter_started == 1)
255  {
256  // adc_data_ready = 1;
257  adc_low_power = 1;
258  }
259  };
260  // пересчет кода АЦП в напряжение канала 5V
261  adc_Values[ADC_CHAN_5V ].addVal((float)adc_dma_buffer[ADC_CHAN_5V ]);
262  if( adc_Values[ADC_CHAN_5V ].calc()){
263  tmp_mv = aprox_Volt( ADC_CHAN_5V );
264 
265  // if(adc_Values[ADC_CHAN_5V ].GetAverage(&tVal)){
266  // tmp_mv = ((u32)tVal * VDD_APPLY) / ADC_VAL_DR_MAX;
267  adc_5V_mv = (u32)( tmp_mv * (3.3f + 5.1f) / 5.1f*1000.0f); // расчет напряжения на делителе, сопротивление в Ом
268  // }
269  };
270  // пересчет кода АЦП в температуру
271  // int32_t tmp = adc_dma_buffer[ADC_CHAN_TEMP];
272  adc_Values[ADC_CHAN_TEMP].addVal((float)adc_dma_buffer[ADC_CHAN_TEMP]);
273  if( adc_Values[ADC_CHAN_TEMP].calc()){
274  tmp_mv = aprox_Volt( ADC_CHAN_TEMP );
275  float term =(ADC_TEMP_BASE - tmp_mv) / ADC_TEMP_SLOPE + 25;
277  // int32_t tmp = (((int32_t)tmp_mv * VDD_APPLY / VDD_CALIB) - (int32_t)(*TEMP30_CAL_ADDR));
278  // tmp = tmp * (int32_t)(110 - 30);
279  // tmp = tmp / (int32_t)((*TEMP110_CAL_ADDR) - (*TEMP30_CAL_ADDR));
280  // tmp = tmp + 30;
281  s32 sign = 1;
282  if(term < 0.0f)
283  sign = -1;
284  float intg ;
285  float frct = modff(term , &intg);
286  if((frct * sign) > 0.5f)
287  intg += (1.0f * sign);
288  adc_temperature = (s32)intg;
289  }
